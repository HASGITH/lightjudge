  <p><button onclick="loadTasks()">Показать задачи</button></p>
  <ul id="tasks"></ul>

  <script>
    async function loadTasks() {
      const res = await fetch("https://your-backend.onrender.com/tasks");
      const data = await res.json();
      document.getElementById("tasks").innerHTML =
        data.map(t => `<li>${t.id}. ${t.title}</li>`).join("");
    }
  </script>
